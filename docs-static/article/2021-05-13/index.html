const htmlParse = (htmlText) =>{
  try{
    const dom = new JSDOM(htmlText,{
      runScripts: "dangerously",
      resources: "usable"
    });
    // get element
    setTimeout(()=>{
      const text = dom.window.document.title;
      console.log(`text:`,text);
    },1000);
  }catch (e){
    console.error(e)
  }

  htmls.forEach((html) => {
    htmlParse(html)
