const builder = new xml2js.Builder();

const urls = ["site1","site2","site3"];

const sitemap = {
  urlset:{
    $:{

    },
    url:[]
  }
};

  (v)=>{
    sitemap.urlset.url.push({
      loc:v
    })
  }
);


var xml = builder.buildObject(sitemap);
