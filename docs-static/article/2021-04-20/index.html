const xml2js = require('xml2js');
const builder = new xml2js.Builder();

const urls = ["site1","site2","site3"];

const sitemap = {
  urlset:{
    $:{
      "xmlns": "http://www.sitemaps.org/schemas/sitemap/0.9"
    },
    url:[]
  }
};

urls.forEach(
  (v)=>{
    sitemap.urlset.url.push({
      loc:v
    })
  }
);


var xml = builder.buildObject(sitemap);

console.log(xml);
